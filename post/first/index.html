duration操作

    using namespace std::chrono;
    duration<int, std::ratio<1>> sec1(65); // 秒

    std::cout << seconds(sec1).count() << "秒\n"
              << milliseconds(sec1).count() << "毫秒\n"
              << microseconds(sec1).count() << "微秒\n"
              << nanoseconds(sec1).count() << "纳秒\n";

    // 精度损失，需要转型
    std::cout << duration_cast<minutes>(sec1).count() << "分钟"
              << duration_cast<hours>(sec1).count() << "小时";

    using namespace std::chrono;
    system_clock::time_point                                        timep = system_clock::now();
    std::chrono::time_point<system_clock, std::chrono::nanoseconds> b     = system_clock::now();

    auto n = timep.time_since_epoch();
    std::cout << "1970.1.1至现在经过的纳秒: " << n.count() << "\n";
    seconds sec = duration_cast<seconds>(n);
    std::cout << "1970.1.1至现在经过的秒: " << sec.count() << "\n";
    minutes m = duration_cast<minutes>(sec);
    std::cout << "1970.1.1至现在经过的分钟: " << m.count() << "\n";
    hours h = duration_cast<hours>(sec);
    std::cout << "1970.1.1至现在经过的小时: " << h.count() << "\n";

    auto        tp    = system_clock::from_time_t(sec.count());
    std::time_t timet = system_clock::to_time_t(tp);
    tm          *t    = std::localtime(&timet);
    std::cout << "now time: " << std::put_time(t, "%F %T") << "\n";

    using namespace std::chrono;

    time_point<system_clock, nanoseconds> unix_pass = system_clock::now();

    nanoseconds unix_nano = unix_pass.time_since_epoch();

    using day = duration<int32_t, std::ratio<60 * 60 * 24>>;
    using month = duration<int32_t, std::ratio<60 * 60 * 24 * 30>>;
    using year = duration<int32_t, std::ratio<60 * 60 * 24 * 30 * 12>>;

    std::cout << "从1970.1.1开始经过了";

    year unix_year = duration_cast<year>(unix_nano);
    std::cout << unix_year.count() << "年\n";
    unix_nano = unix_nano - unix_year;

    month unix_month = duration_cast<month>(unix_nano);
    std::cout << unix_month.count() << "月\n";
    unix_nano = unix_nano - unix_month;

    day unix_day = duration_cast<day>(unix_nano);
    std::cout << unix_day.count() << "天\n";
    unix_nano = unix_nano - unix_day;

    hours unix_hour = duration_cast<hours>(unix_nano);
    std::cout << unix_hour.count() << "小时\n";
    unix_nano = unix_nano - unix_hour;

    minutes unix_min = duration_cast<minutes>(unix_nano);
    std::cout << unix_min.count() << "分钟\n";
    unix_nano = unix_nano - unix_min;

    seconds unix_sec = duration_cast<seconds>(unix_nano);
    std::cout << unix_sec.count() << "秒\n";
